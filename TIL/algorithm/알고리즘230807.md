```python
# int() 와 같은 atoi() 함수
def atoi(s):
	i = 0
	for x in s:
		i = i*10 + ord(x) - ord('0')
	return i
```

```python
# str() 와 같은 itoa() 함수
def itoa(i):
	a = ''
	while i > 0:
		a = chr(i % 10 + ord('0')) + a
		i //= 10
	return a
```
